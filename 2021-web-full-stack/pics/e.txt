int delete(struct array *arr , int index)
{

    int x=0;
    int i;
    if(index>=0 && index < arr->length)
    {
        x=arr->A[index];
        for(i=index ; i<arr->length-1 ;i++);
        {
            arr->A[i]=arr->A[i+1];
        }
        arr->length--;
        return x;
    }
    return 0;
}
